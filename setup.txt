# ==================================
#  Python Project Setup Instructions
# ==================================

# These commands should be run in your project's root directory (e.g., music_recommender/) from your terminal.

# --- STEP 1: Create the Virtual Environment ---
# This creates an isolated environment folder named 'venv' for your project's packages.
# You only need to do this once per project.

python -m venv venv


# --- STEP 2: Activate the Virtual Environment ---
# This "turns on" the isolated environment. You must do this every time you open a new terminal to work on the project.
# Your terminal prompt will change to show `(venv)` at the beginning.

# On Windows (Command Prompt or PowerShell):
venv\Scripts\activate

# On macOS or Linux:
source venv/bin/activate


# --- STEP 3: Install Required Packages ---
# With the venv active, now you can install your packages. They will be installed
# inside the 'venv' folder, not globally on your computer.

pip install requests python-dotenv


# --- STEP 4: Create the requirements.txt File ---
# This command automatically generates a list of all packages and their exact versions
# that are currently installed in your active environment.

pip freeze > requirements.txt


# --- HOW TO USE requirements.txt IN THE FUTURE ---
# If you move this project to a new computer or share it, you can instantly
# install all the correct packages by running the following command after activating the venv:

pip install -r requirements.txt
